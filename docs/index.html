context7 validate --with-score

context7 serve

üìö API Reference & Integration

Programmatic Usage

MCP Server Integration
import { Context7MCPServer } from '@wtthornton/context7-mcp';

// Start MCP server with auto-detection
const server = new Context7MCPServer({
  projectRoot: './my-project',
  projectType: 'react-webapp', // auto-detected if not specified
  mcpEnabled: true
});

await server.start();
console.log('MCP Server running for AI assistants');
Project Quality Scoring
import { ProjectScorer } from '@wtthornton/context7-mcp';

// Advanced quality analysis
const scorer = new ProjectScorer({
  projectRoot: './my-project',
  verbose: true
});

const results = await scorer.scoreProject({
  categories: ['structure', 'quality', 'testing', 'security'],
  detailed: true,
  recommendations: true
});

console.log(`Overall Score: ${results.overall.score}/100 (${results.overall.grade})`);

// Access detailed metrics
results.categories.quality.details.eslintErrors; // Real ESLint error count
results.categories.security.details.vulnerabilities; // npm audit results
results.categories.testing.details.coverage; // Real coverage metrics
Validation and Testing
import { Context7Validator, MCPConnectionTester } from '@wtthornton/context7-mcp';

// Project validation
const validator = new Context7Validator({
  projectRoot: './my-project',
  strictMode: true
});

const validation = await validator.runValidation();
console.log(`Compliance: ${validation.report.successRate}%`);

// MCP functionality testing
const tester = new MCPConnectionTester({
  serverPath: 'src/mcp-server.js',
  timeout: 30000
});

const testResults = await tester.runTests();
console.log(`MCP Tests: ${testResults.report.passed}/${testResults.report.total} passed`);

Configuration Options

context7.config.js
export default {
  // Project Configuration
  projectName: 'my-awesome-app',
  projectType: 'react-webapp', // react-webapp, vue-webapp, node-api, etc.
  
  // MCP Server Settings
  mcp: {
    enabled: true,
    transport: 'stdio', // stdio, http, websocket
    port: 3001,
    timeout: 30000,
    capabilities: ['resources', 'tools', 'prompts']
  },
  
  // Validation Configuration
  validation: {
    strictMode: true,
    autoFix: false,
    requiredFiles: ['AGENTS.md', 'CLAUDE.md'],
    patterns: {
      components: 'PascalCase',
      hooks: 'camelCase',
      files: 'kebab-case'
    }
  },
  
  // Scoring Configuration
  scoring: {
    categories: ['all'], // or specific categories
    thresholds: {
      excellent: 90,
      good: 80,
      acceptable: 70
    },
    weights: {
      structure: 20,
      quality: 20,
      performance: 15,
      testing: 15,
      security: 15,
      developerExperience: 10,
      completeness: 5
    }
  },
  
  // Custom Patterns
  patterns: {
    component: './custom-patterns/component.tsx',
    hook: './custom-patterns/hook.ts',
    service: './custom-patterns/service.ts'
  }
};
